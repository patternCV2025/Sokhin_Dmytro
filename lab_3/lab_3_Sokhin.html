import numpy as np
import matplotlib.pyplot as plt

#  
x = np.arange(0, 10.1, 0.1)
y = np.sin(2 * x) + 0.1 * (x ** 2)  # : sin(2x) + 0.1*x^2

#   :   
x_train = x[::2]
y_train = y[::2]

#    : N  (  N-1)
N = 13  #   12- 
L = 20  #  

#      x
X = np.array([[a ** n for n in range(N)] for a in x])
#      
X_train = X[::2]
Y = y_train  #   

#   : lambda*I,     
IL = np.diag([0] + [L] * (N - 1))

#    :
# w = (X_train^T * X_train + lambda*I)^(-1) * X_train^T * Y
w = np.linalg.inv(X_train.T @ X_train + IL) @ (X_train.T @ Y)
print(" :")
print(w)

#         x
yy = np.array([np.dot(w, row) for row in X])

#  :     
plt.figure(figsize=(10, 6))
plt.plot(x, y, label=" ", linewidth=2)

plt.scatter(x_train, y_train, color="red", label=" ")

import json
import numpy as np
import matplotlib.pyplot as plt

#       
with open("data_variant10.json", "r", encoding="utf-8") as f:
    data = json.load(f)
    
#  : x_train_10   , y_train_10   
x_train = data["x_train_10"]
y_train = data["y_train_10"]

#     (  )  :
#         : 10*x1, 10*x2, 5*(x1+x2)
x_train = [x + [10 * x[0], 10 * x[1], 5 * (x[0] + x[1])] for x in x_train]
x_train = np.array(x_train)
y_train = np.array(y_train)

# ---       ( L1-) ---

#    (   w*x  y,   )
def loss(w, x, y):
    M = np.dot(w, x) * y
    return 2 / (1 + np.exp(M))

#      w  L1-
def df(w, x, y):
    L1 = 1.0  #  L1- (     )
    M = np.dot(w, x) * y
    #        L1-
    return -2 * (1 + np.exp(M))**(-2) * np.exp(M) * x * y + L1 * np.sign(w)

# ---    ---

n_features = len(x_train[0])  #    (   5)
n_train = len(x_train)        #   
w = np.zeros(n_features)      #    ( )

nt = 0.00001  #   (learning rate)
lm = 0.01     #  ""    
N = 5000      #      (SGD)

#      ( )
Q = np.mean([loss(w, x, y) for x, y in zip(x_train, y_train)])
Q_plot = [Q]

# ---    (SGD) ---
for i in range(N):
    k = np.random.randint(0, n_train)         #      
    ek = loss(w, x_train[k], y_train[k])        #     
    w = w - nt * df(w, x_train[k], y_train[k])    #     SGD  L1-
    Q = lm * ek + (1 - lm) * Q                  #    
    Q_plot.append(Q)

#     
Q_final = np.mean([loss(w, x, y) for x, y in zip(x_train, y_train)])
print(" :", w)
print("  :", Q_final)

# ---    ---
plt.figure(figsize=(10, 5))
plt.plot(Q_plot, label=" ")
plt.xlabel("")
plt.ylabel("")
plt.title("    ")
plt.legend()
plt.grid(True)
plt.show()

#   :      :
# 10*x1, 10*x2, 5*(x1+x2)
x_train = [x + [10 * x[0], 10 * x[1], 5 * (x[0] + x[1])] for x in x_train]
x_train = np.array(x_train)
y_train = np.array(y_train)

# ---        L2- ---

def loss(w, x, y):
    """
      :    (wx)*y,   .
    """
    M = np.dot(w, x) * y
    return 2 / (1 + np.exp(M))

def df(w, x, y):
    """
         w   L2-.
     L2-      2 * L2 * w.
    """
    L2 = 1.0  #  L2- (     )
    M = np.dot(w, x) * y
    grad_loss = -2 * (1 + np.exp(M))**(-2) * np.exp(M) * x * y
    grad_reg = 2 * L2 * w
    return grad_loss + grad_reg

# ---    ---

n_features = len(x_train[0])  #   (   5)
n_train = len(x_train)        #   
w = np.zeros(n_features)      #   

nt = 0.00001  #   (learning rate)
lm = 0.01     #     
N = 5000      #      (SGD)

#    
Q = np.mean([loss(w, x, y) for x, y in zip(x_train, y_train)])
Q_plot = [Q]

# ---    (SGD)  L2- ---

for i in range(N):
    k = np.random.randint(0, n_train)  #      
    ek = loss(w, x_train[k], y_train[k])  #     
    w = w - nt * df(w, x_train[k], y_train[k])  #  
    Q = lm * ek + (1 - lm) * Q  #    ()
    Q_plot.append(Q)

#    
Q_final = np.mean([loss(w, x, y) for x, y in zip(x_train, y_train)])
print("  (L2-):", w)
print("   (L2-):", Q_final)

import numpy as np

#   
# x_train    ( bias), y_train    {1, -1}
x_train = np.array([[0.5, 1.0],
                    [1.5, 2.0],
                    [3.0, 4.0],
                    [2.5, 3.0]])
y_train = np.array([1, 1, -1, -1])

#     bias
X = np.hstack([np.ones((x_train.shape[0], 1)), x_train])

#    (bias + 2 )
w = np.zeros(X.shape[1])

#  
learning_rate = 0.01
n_iterations = 1000

#    (SGD)  
for i in range(n_iterations):
    #    
    idx = np.random.randint(0, X.shape[0])
    xi = X[idx]
    yi = y_train[idx]
    
    #    ,  
    if yi * np.dot(w, xi) <= 0:
        w = w + learning_rate * yi * xi

print(" :", w)

#   
def predict(X, w):
    return np.where(np.dot(X, w) >= 0, 1, -1)

#    
predictions = predict(X, w)
print(" :", predictions)
